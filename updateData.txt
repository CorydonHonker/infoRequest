Set fso = CreateObject("Scripting.FileSystemObject")
Const ForReading = 1
Const ForWriting = 2
Const ForAppending = 8
strFile = "LogFile.log"
memorySpace = 16
'====Read===='
'==Check==='
If Not (fso.FileExists(strFile)) Then
	outFile= strFile
	Set objTextFile = fso.CreateTextFile(outFile,True)
	objTextFile.Write "00=00"
	objTextFile.Close
End If

Set objTextFile = fso.OpenTextFile(strFile, ForReading)
objTextFile.ReadAll
if objTextFile.Line < memorySpace Then
	Set objTextFile = fso.OpenTextFile(strFile, ForAppending)
	for i = objTextFile.Line to memorySpace
		objTextFile.WriteLine
		objTextFile.Write right("0"&i&"=00",5)
	Next
objTextFile.Close
end if
'==-----==='
Function getLineData(filename,line)
if line > memorySpace+1 Then
	WScript.echo line
	Exit function
end if
	Set f = fso.OpenTextFile(filename)
	Do Until f.AtEndOfStream
		if f.Line = line Then
			getLineData = f.ReadLine
		else
			f.ReadLine
		end if
	Loop
f.Close
end function

Function LoadStringFromFile(filename)
    Dim fso, f
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set f = fso.OpenTextFile(filename, ForReading)
    LoadStringFromFile = f.ReadAll
    f.Close
End Function

'====----===='
'====Write==='
Sub SaveStringToFile(filename, text)
    Dim fso, f
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set f = fso.OpenTextFile(filename, ForWriting)
    f.Write text
    f.Close
End Sub

function replaceData(myTargetStr,myMessage)
	data= LoadStringFromFile(strFile)
	replaceData = Replace(data,myTargetStr,myMessage)
	SaveStringToFile strFile, replaceData
end function
